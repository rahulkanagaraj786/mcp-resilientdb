41def get_changed_files(base_ref: str = "HEAD^") -> set[str]:
42 """Get list of changed files between base_ref and HEAD."""
43 try:
44 # Get diff between base_ref and HEAD
45 result = subprocess.run(
46 ["git", "diff", "--name-only", base_ref, "HEAD"],
47 cwd=REPO_ROOT,
48 capture_output=True,
49 text=True,
50 check=True,
51 )
52 changed = set(result.stdout.strip().split("\n"))
53 # Filter out empty strings
54 changed.discard("")
55 return changed
56 except subprocess.CalledProcessError as exc:
57 raise SystemExit(f"git diff failed: {exc}") from exc

60def should_exclude_path(path: str) -> bool:
61 """Check if a path should be excluded from triggering updates."""
62 for pattern in EXCLUDE_PATTERNS:
63 # Simple pattern matching (supports ** for recursive)
64 if pattern.endswith("/**"):
65 prefix = pattern[:-3] # Remove /**
66 if path.startswith(prefix + "/") or path == prefix:
67 return True
68 elif path == pattern or path.startswith(pattern + "/"):
69 return True
70 return False

73def detect_changed_tools(base_ref: str = "HEAD^") -> list[str]:
74 """
75 Detect which tools have changed by checking if their code_root paths
76 have any changed files.
77 """
78 tool_map = load_tool_map()
79 changed_files = get_changed_files(base_ref)
80
81 # Filter out excluded paths
82 relevant_changes = {
83 f for f in changed_files if not should_exclude_path(f)
84 }
85
86 if not relevant_changes:
87 return []
88
89 # Find which tools have changes in their code_root
90 changed_tools = []
91 for entry in tool_map:
92 tool_name = entry.get("name")
93 code_root = entry.get("code_root")
94
95 if not code_root or not tool_name:
96 continue
97
98 # Check if any changed file is under this tool's code_root
99 for changed_file in relevant_changes:
100 if changed_file.startswith(code_root + "/") or changed_file == code_root:
101 changed_tools.append(tool_name)
102 break # Only add once per tool
103
104 return changed_tools
