71bool PoE::ReceivePrepare(std::unique_ptr<Proposal> proposal) {
72 std::unique_ptr<Transaction> txn = nullptr;
73 {
74 std::unique_lock<std::mutex> lk(mutex_);
75 received_[proposal->hash()].insert(proposal->proposer());
76 auto it = data_.find(proposal->hash());
77 if (it != data_.end()) {
78 if (received_[proposal->hash()].size() >= 2 * f_ + 1) {
79 txn = std::move(it->second);
80 data_.erase(it);
81 }
82 }
83 }
84 if (txn != nullptr) {
85 commit_(*txn);
86 }
87 return true;
88}
