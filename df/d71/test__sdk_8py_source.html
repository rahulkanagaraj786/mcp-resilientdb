21from resdb_driver import Resdb
22
23db_root_url = "http://127.0.0.1:18000"
24
25db = Resdb(db_root_url)
26from resdb_driver.crypto import generate_keypair
27
28alice, bob = generate_keypair(), generate_keypair()
29#%%
30
31# create a digital asset for Alice
32game_boy_token = {
33 "data": {
34 "token_for": {"game_boy": {"serial_number": "LR35902"}},
35 "description": "Time share token. Each token equals one hour of usage.",
36 },
37}
38
39#%%
40
41# prepare the transaction with the digital asset and issue 10 tokens for Bob
42prepared_token_tx = db.transactions.prepare(
43 operation="CREATE",
44 signers=alice.public_key,
45 recipients=[([bob.public_key], 10)],
46 asset=game_boy_token,
47)
48
49#%%
50# fulfill the tnx
51fulfilled_token_tx = db.transactions.fulfill(
52 prepared_token_tx, private_keys=alice.private_key
53)
54
55#%%
56db.transactions.send_commit(fulfilled_token_tx)
57
58#%%
59transfer_asset = {"id": fulfilled_token_tx["id"]}
60output_index = 0
61output = fulfilled_token_tx["outputs"][output_index]
62transfer_input = {
63 "fulfillment": output["condition"]["details"],
64 "fulfills": {"output_index": output_index, "transaction_id": transfer_asset["id"]},
65 "owners_before": output["public_keys"],
66}
67
68#%%
69
70# Bob transfers 5 tokens to Alice, and is left with 5 tokens
71# The sums of amounts in recipients must match the input amount (10)
72prepared_transfer_tx = db.transactions.prepare(
73 operation="TRANSFER",
74 asset=transfer_asset,
75 inputs=transfer_input,
76 recipients=[([alice.public_key], 5), ([bob.public_key], 5)],
77)
78
79#%%
80
81fulfilled_transfer_tx = db.transactions.fulfill(
82 prepared_transfer_tx, private_keys=bob.private_key
83)
84
85#%%
86
87sent_transfer_tx = db.transactions.send_commit(fulfilled_transfer_tx)
88
89
90# %%
91
92db.transactions.retrieve(txid=sent_transfer_tx)
