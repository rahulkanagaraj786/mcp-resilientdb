35bool IsValidAddress(const std::string& address) {
36 if (address.length() != 42) {
37 return false;
38 }
39
40 if (address.substr(0, 2) != "0x") {
41 return false;
42 }
43
44 for (size_t i = 2; i < address.length(); ++i) {
45 if (!std::isxdigit(address[i])) {
46 return false;
47 }
48 }
49 return true;
50}

99nlohmann::json ReadJSConfig(const std::string& config_path) {

116int main(int argc, char** argv) {
117 if (argc < 2) {
118 printf("<cmd> -c [config]\n");
119 return 0;
120 }
121
122 std::string config_file;
123
124 std::string cmd;
125 std::string caller_address, contract_name, contract_path, params,
126 contract_address, func_name;
127 int c;
128 int option_index;
129 std::string client_config_file;
130 while ((c = getopt_long(argc, argv, "c:h", long_options, &option_index)) !=
131 -1) {
132 switch (c) {
133 case -1:
134 break;
135 case 'f':
136 config_file = optarg;
137 break;
138 case 'c':
139 client_config_file = optarg;
140 break;
141 case 'h':
142 ShowUsage();
143 break;
144 }
145 }
146
147 nlohmann::json js = ReadJSConfig(config_file);
148 std::cout << js << std::endl;
149 cmd = GetValue(js, "command");
150
151 ResDBConfig config = GenerateResDBConfig(client_config_file);
152 config.SetClientTimeoutMs(100000);
153
154 ContractClient client(config);
155 printf("cmd = %s\n", cmd.c_str());
156 if (cmd == "create_account") {
157 CreateAccount(&client);
158 } else if (cmd == "deploy") {
159 contract_path = GetValue(js, "contract_path");
160 contract_name = GetValue(js, "contract_name");
161 contract_address = GetValue(js, "owner_address");
162 params = GetValue(js, "init_params");
163
164 if (!IsValidAddress(contract_address)) {
165 printf("ERROR: Invalid owner address format: %s\n",
166 contract_address.c_str());
167 return 1;
168 }
169
170 printf(
171 "contract path %s cmd %s contract name %s caller_address %s init "
172 "params %s\n",
173 contract_path.c_str(), cmd.c_str(), contract_name.c_str(),
174 contract_address.c_str(), params.c_str());
175
176 std::vector<std::string> init_params;
177 boost::split(init_params, params, boost::is_any_of(","));
178
179 DeployContract(&client, contract_address, contract_name, contract_path,
180 init_params);
181 } else if (cmd == "execute") {
182 caller_address = GetValue(js, "caller_address");
183 contract_address = GetValue(js, "contract_address");
184 func_name = GetValue(js, "func_name");
185 params = GetValue(js, "params");
186
187 if (!IsValidAddress(caller_address)) {
188 printf("ERROR: Invalid caller address format: %s\n",
189 caller_address.c_str());
190 return 1;
191 }
192
193 if (!IsValidAddress(contract_address)) {
194 printf("ERROR: Invalid contract address format: %s\n",
195 contract_address.c_str());
196 return 1;
197 }
198
199 printf(
200 "execute\n caller address:%s\n contract address: %s\n func: %s\n "
201 "params:%s\n",
202 caller_address.c_str(), contract_address.c_str(), func_name.c_str(),
203 params.c_str());
204 std::vector<std::string> func_params;
205 boost::split(func_params, params, boost::is_any_of(","));
206
207 ExecuteContract(&client, caller_address, contract_address, func_name,
208 func_params);
209 } else if (cmd == "get_balance") {
210 std::string address = GetValue(js, "address");
211
212 // Validate address format
213 if (!IsValidAddress(address)) {
214 printf("ERROR: Invalid address format: %s\n", address.c_str());
215 return 1;
216 }
217
218 auto balance_or = client.GetBalance(address);
219 printf("get address %s balance %s\n", address.c_str(),
220 (*balance_or).c_str());
221 } else if (cmd == "set_balance") {
222 std::string address = GetValue(js, "address");
223 std::string balance = GetValue(js, "balance");
224
225 // Validate address format
226 if (!IsValidAddress(address)) {
227 printf("ERROR: Invalid address format: %s\n", address.c_str());
228 return 1;
229 }
230 printf("address %s balance %s\n", address.c_str(), balance.c_str());
231 auto ret = client.SetBalance(address, balance);
232 printf("set address %s balance %s ret %s\n", address.c_str(),
233 balance.c_str(), (*ret).c_str());
234 }
235}
