40KeyInfo ReadKey(const std::string& file_name) {
41 int fd = open(file_name.c_str(), O_RDONLY, 0666);
42 if (fd < 0) {
43 LOG(ERROR) << "open file:" << file_name << " fail:" << strerror(errno);
44 }
45 assert(fd >= 0);
46
47 std::string res;
48 int read_len = 0;
49 char tmp[1024];
50 while (true) {
51 read_len = read(fd, tmp, sizeof(tmp));
52 if (read_len <= 0) {
53 break;
54 }
55 res.append(tmp, read_len);
56 }
57 close(fd);
58 KeyInfo key;
59 assert(key.ParseFromString(res));
60 return key;
61}

103ResConfigData ReadConfigFromFile(const std::string& file_name) {
104 std::stringstream json_data;
105 std::ifstream infile(file_name.c_str());
106 if (!infile.is_open()) {
107 std::cerr << "Failed to open file." << file_name << " " << strerror(errno)
108 << std::endl;
109 return ResConfigData();
110 }
111
112 json_data << infile.rdbuf();
113 std::string cleanJson = RemoveJsonComments(json_data.str());
114
115 ResConfigData config_data;
116 JsonParseOptions options;
117 auto status = JsonStringToMessage(cleanJson, &config_data, options);
118 if (!status.ok()) {
119 LOG(ERROR) << "parse json :" << file_name << " fail:" << status.message();
120 }
121 assert(status.ok());
122 return config_data;
123}

125std::vector<ReplicaInfo> ReadConfig(const std::string& file_name) {
126 std::vector<ReplicaInfo> replicas;
127 std::stringstream json_data;
128 std::ifstream infile(file_name.c_str());
129 if (!infile.is_open()) {
130 std::cerr << "Failed to open file." << file_name << " " << strerror(errno)
131 << std::endl;
132 return replicas;
133 }
134
135 json_data << infile.rdbuf();
136 std::string cleanJson = RemoveJsonComments(json_data.str());
137
138 RegionInfo region_info;
139 JsonParseOptions options;
140 auto status = JsonStringToMessage(cleanJson, &region_info, options);
141 if (!status.ok()) {
142 LOG(ERROR) << "parse json :" << file_name << " fail:" << status.message();
143 }
144 assert(status.ok());
145 for (const auto& replica_info : region_info.replica_info()) {
146 LOG(ERROR) << "parse json id:" << replica_info.id()
147 << " ip:" << replica_info.ip()
148 << " port:" << replica_info.port();
149 replicas.push_back(GenerateReplicaInfo(replica_info.id(), replica_info.ip(),
150 replica_info.port()));
151 }
152 return replicas;
153}

155std::unique_ptr<ResDBConfig> GenerateResDBConfig(
156 const std::string& config_file, const std::string& private_key_file,
157 const std::string& cert_file, std::optional<ReplicaInfo> self_info,
158 std::optional<ConfigGenFunc> gen_func) {
159 ResConfigData config_data = ReadConfigFromFile(config_file);
160 KeyInfo private_key = ReadKey(private_key_file);
161 CertificateInfo cert_info = ReadCert(cert_file);
162
163 LOG(ERROR) << "private key:" << private_key.DebugString()
164 << " cert:" << cert_info.DebugString();
165 if (!self_info.has_value()) {
166 self_info = ReplicaInfo();
167 }
168
169 (*self_info).set_id(cert_info.public_key().public_key_info().node_id());
170 (*self_info).set_ip(cert_info.public_key().public_key_info().ip());
171 (*self_info).set_port(cert_info.public_key().public_key_info().port());
172
173 *(*self_info).mutable_certificate_info() = cert_info;
174
175 if (gen_func.has_value()) {
176 return (*gen_func)(config_data, self_info.value(), private_key, cert_info);
177 }
178 return std::make_unique<ResDBConfig>(config_data, self_info.value(),
179 private_key, cert_info);
180}
