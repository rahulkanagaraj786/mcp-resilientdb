33bool RsaVerifyString(const std::string& message, const std::string& public_key,
34 const std::string& signature) {
35 try {
36 // decode and load public key (using pipeline)
37 CryptoPP::RSA::PublicKey rsa_public_key;
38 rsa_public_key.Load(
39 CryptoPP::StringSource(public_key, true, new CryptoPP::HexDecoder())
40 .Ref());
41
42 // decode signature
43 std::string decoded_signature;
44 CryptoPP::StringSource ss(
45 signature, true,
46 new CryptoPP::HexDecoder(new CryptoPP::StringSink(decoded_signature)));
47
48 // verify message
49 bool result = false;
50 CryptoPP::RSASS<CryptoPP::PSSR, CryptoPP::Whirlpool>::Verifier verifier(
51 rsa_public_key);
52 CryptoPP::StringSource ss2(
53 decoded_signature + message, true,
54 new CryptoPP::SignatureVerificationFilter(
55 verifier,
56 new CryptoPP::ArraySink((CryptoPP::byte*)&result, sizeof(result))));
57
58 return result;
59 } catch (...) {
60 LOG(ERROR) << "key not valid";
61 return false;
62 }
63}

95std::string RsaSignString(const std::string& private_key,
96 const std::string& message) {
97 // decode and load private key (using pipeline)
98 CryptoPP::RSA::PrivateKey rsa_private_key;
99 rsa_private_key.Load(
100 CryptoPP::StringSource(private_key, true, new CryptoPP::HexDecoder())
101 .Ref());
102
103 // sign message
104 std::string signature;
105 CryptoPP::RSASS<CryptoPP::PSSR, CryptoPP::Whirlpool>::Signer signer(
106 rsa_private_key);
107 CryptoPP::AutoSeededRandomPool rng;
108 CryptoPP::StringSource ss(
109 message, true,
110 new CryptoPP::SignerFilter(
111 rng, signer,
112 new CryptoPP::HexEncoder(new CryptoPP::StringSink(signature))));
113 return signature;
114}

116std::string ECDSASignString(const std::string& private_key,
117 const std::string& message) {
118 try {
119 CryptoPP::ECDSA<CryptoPP::ECP, CryptoPP::SHA256>::PrivateKey privateKey;
120 privateKey.Load(
121 CryptoPP::StringSource(private_key, true, new CryptoPP::HexDecoder())
122 .Ref());
123 CryptoPP::AutoSeededRandomPool prng;
124 std::string signature;
125
126 CryptoPP::StringSource ss1(
127 message, true /*pump all*/,
128 new CryptoPP::SignerFilter(
129 prng,
130 CryptoPP::ECDSA<CryptoPP::ECP, CryptoPP::SHA256>::Signer(
131 privateKey),
132 new CryptoPP::StringSink(signature)));
133
134 std::string output;
135 CryptoPP::StringSource(signature, true,
136 new CryptoPP::HexEncoder(
137 new CryptoPP::StringSink(output)) // HexEncoder
138 ); // StringSource
139 return output;
140 } catch (...) {
141 LOG(ERROR) << "sign error";
142 return "";
143 }
