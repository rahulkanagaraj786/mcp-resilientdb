19"""Utility script to generate Ed25519 keypairs for ResilientDB.
20
21This is a standalone utility script for manual key generation.
22For automated key generation, use the MCP server's generateKeys tool instead.
23"""
24import sys
25from collections import namedtuple
26
27try:
28 from cryptoconditions import crypto as _cc_crypto
29except ImportError:
30 print("Error: Could not import cryptoconditions.")
31 print("Install it with: pip install cryptoconditions")
32 sys.exit(1)
33
34CryptoKeypair = namedtuple("CryptoKeypair", ("private_key", "public_key"))
35
36
37def generate_keypair() -> CryptoKeypair:
38 """Generate an Ed25519 keypair (base58-encoded, ResilientDB format)."""
39 priv, pub = (k.decode() for k in _cc_crypto.ed25519_generate_key_pair())
40 return CryptoKeypair(private_key=priv, public_key=pub)
41
42
43# Generate keypairs
44signer = generate_keypair()
45recipient = generate_keypair()
46
