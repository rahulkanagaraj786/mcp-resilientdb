37Consensus::Consensus(const ResDBConfig& config,
38 std::unique_ptr<TransactionManager> executor)
39 : common::Consensus(config, std::move(executor)) {
40 int total_replicas = config_.GetReplicaNum();
41 int f = (total_replicas - 1) / 3;
42
43 if (config_.GetPublicKeyCertificateInfo()
44 .public_key()
45 .public_key_info()
46 .type() == CertificateKeyInfo::CLIENT) {
47 SetPerformanceManager(GetPerformanceManager());
48 }
49
50 Init();
51
52 start_ = 0;
53
54 if (config_.GetPublicKeyCertificateInfo()
55 .public_key()
56 .public_key_info()
57 .type() != CertificateKeyInfo::CLIENT) {
58 poe_ = std::make_unique<PoE>(config_.GetSelfInfo().id(), f, total_replicas,
59 GetSignatureVerifier());
60 InitProtocol(poe_.get());
61 }
62}

64int Consensus::ProcessCustomConsensus(std::unique_ptr<Request> request) {
65 if (request->user_type() == MessageType::Propose) {
66 std::unique_ptr<Transaction> txn = std::make_unique<Transaction>();
67 if (!txn->ParseFromString(request->data())) {
68 assert(1 == 0);
69 LOG(ERROR) << "parse proposal fail";
70 return -1;
71 }
72 poe_->ReceivePropose(std::move(txn));
73 return 0;
74 } else if (request->user_type() == MessageType::Prepare) {
75 std::unique_ptr<Proposal> proposal = std::make_unique<Proposal>();
76 if (!proposal->ParseFromString(request->data())) {
77 LOG(ERROR) << "parse proposal fail";
78 assert(1 == 0);
79 return -1;
80 }
81 poe_->ReceivePrepare(std::move(proposal));
82 return 0;
83 }
84 return 0;
85}
