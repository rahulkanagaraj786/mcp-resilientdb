211async def fetch_repo_tree(owner: str, repo: str, branch: str = "main") -> List[dict]:
212 """Fetch repository tree with better branch handling"""
213 headers = get_auth_headers()
214
215 # Try both main and master branches
216 branches_to_try = [branch]
217 if branch == "main":
218 branches_to_try.append("master")
219 elif branch == "master":
220 branches_to_try.append("main")
221
222 async with httpx.AsyncClient(follow_redirects=True, timeout=30.0) as client:
223 last_error = None
224 for try_branch in branches_to_try:
225 try:
226 url = f"https://api.github.com/repos/{owner}/{repo}/git/trees/{try_branch}?recursive=1"
227 response = await client.get(url, headers=headers)
228 response.raise_for_status()
229 data = response.json()
230 return [item for item in data.get("tree", []) if item['type'] == 'blob']
231 except Exception as e:
232 last_error = e
233 continue
234
235 # If all branches failed, raise the last error
236 if last_error:
237 raise last_error
238 return []

259# Alternate parser using Python's built-in ast module
260def parse_python_functions_ast(code: str) -> List[str]:
261 functions = []
262 try:
263 tree = ast.parse(code)
264 for node in ast.walk(tree):
265 if isinstance(node, ast.FunctionDef):
266 # Generate a simple string summary (function signature)
267 args = [arg.arg for arg in node.args.args]
268 arglist = ", ".join(args)
269 functions.append(f"def {node.name}({arglist}):")
270 except Exception:
271 # If parsing fails, return empty list to avoid crashing MCP server
272 return []
273 return functions

278# Helper function for Contents API recursion (NOT an MCP tool)
279async def _fetch_files_recursive(
280 owner: str,
281 repo: str,
282 branch: str,
283 path: str = ""
284) -> List[str]:
285 """
286 Internal helper to recursively fetch files using Contents API.
287 """
288 headers = get_auth_headers()
289 files = []
290 url = f"https://api.github.com/repos/{owner}/{repo}/contents/{path}?per_page=100"
291
292 async with httpx.AsyncClient(follow_redirects=True, timeout=30.0) as client:
293 while url:
294 try:
295 response = await client.get(url, headers=headers)
296 response.raise_for_status()
297 except Exception as e:
298 print(f"Error fetching {url}: {e}")
299 break
300
301 # Handle Link header pagination
302 link_header = response.headers.get('Link', '')
303 next_url = None
304 if 'rel="next"' in link_header:
305 next_url = [link.split(';')[0].strip('<> \t') for link in link_header.split(',')
306 if 'rel="next"' in link][0]
307
308 data = response.json()
309
310 # Single file response (base64 content)
311 if isinstance(data, dict) and 'content' in data:
312 files.append(data['path'])
313 break
314
315 # Directory listing (list of items)
316 for item in data:
317 if item['type'] == 'file':
318 files.append(item['path'])
319 elif item['type'] == 'dir':
320 # Call the HELPER function recursively
321 dir_files = await _fetch_files_recursive(
322 owner, repo, branch, item['path']
323 )
324 files.extend(dir_files)
325
326 url = next_url # Continue pagination
327
328 return files

332@mcp.tool(name="list_github_repo_files")
333async def list_github_repo_files(
334 owner: str,
335 repo: str,
336 branch: str = "main",
337 path: str = ""
338) -> List[str]:
339 """
340 This tool fetches the list of files in a particular repository using the github api mentioned in the MCP tool and not from any online source.
341 """
342 headers = get_auth_headers()
343
344 # FAST PATH: Try Tree API first (single request for entire repo)
345 try:
346 async with httpx.AsyncClient(follow_redirects=True, timeout=30.0) as client:
347 # Try to get the branch SHA first for more reliable results
348 try:
349 ref_url = f"https://api.github.com/repos/{owner}/{repo}/git/ref/heads/{branch}"
350 ref_response = await client.get(ref_url, headers=headers)
351 ref_response.raise_for_status()
352 sha = ref_response.json()['object']['sha']
353 except:
354 # Fallback: use branch name directly
355 sha = branch
356
357 # Get the full tree recursively in one call
358 tree_url = f"https://api.github.com/repos/{owner}/{repo}/git/trees/{sha}?recursive=1"
359 tree_response = await client.get(tree_url, headers=headers)
360 tree_response.raise_for_status()
361 data = tree_response.json()
362
363 # Check if the response was truncated (repo too large)
364 if not data.get('truncated', False):
365 # Success! Return all files from tree
366 files = [item['path'] for item in data.get('tree', [])
367 if item['type'] == 'blob']
368 print(f"✓ Tree API: Found {len(files)} files in one request")
369 return files
370 else:
371 # Repo is too large, tree was truncated
372 print(f"⚠ Tree API truncated, falling back to Contents API...")
373
374 except Exception as e:
375 # Tree API failed for some reason
376 print(f"⚠ Tree API failed ({str(e)}), using Contents API...")
377
378 # SLOW PATH: Fall back to Contents API with recursive traversal
379 print(f"→ Using Contents API (may take longer for large repos)...")
380 files = await _fetch_files_recursive(owner, repo, branch, path)
381 print(f"✓ Contents API: Found {len(files)} files")
382 return files

431@mcp.tool(name="Ingest_Repo_Code")
432async def ingest_repo_code(owner: str, repo: str, branch: str = "main") -> str:
433 """
434 Fetch repo files, chunk code, generate embeddings and index them
435 """
436 global index, metadata, index_dimension
437
438 if model is None:
439 return "⚠️ Semantic search not available - SentenceTransformer not installed"
440
441 try:
442 # Fetch all files using the helper function directly
443 files = await _fetch_files_recursive(owner, repo, branch, "")
444
445 if not files:
446 # Try alternative method
447 try:
448 tree = await fetch_repo_tree(owner, repo, branch)
449 files = [item['path'] for item in tree]
450 except:
451 return f"Could not fetch files from {owner}/{repo}. Try checking the branch name."
452
453 # Filter for code files only
454 code_extensions = ['.py', '.js', '.jsx', '.ts', '.tsx', '.java', '.cpp', '.c', '.go', '.rs']
455 code_files = [f for f in files if any(f.endswith(ext) for ext in code_extensions)]
456
457 if not code_files:
458 return f"No code files found in {owner}/{repo}"
459
460 # Collect all chunks and their metadata
461 all_chunks = []
462 all_metadata = []
463
464 for filepath in code_files[:50]: # Limit to first 50 files to avoid timeout
465 try:
466 content = await fetch_raw_file(owner, repo, filepath, branch)
467 if content:
468 # Simple chunking by lines
469 lines = content.split('\n')
470 chunk_size = 50
471
472 for i in range(0, len(lines), chunk_size):
473 chunk = '\n'.join(lines[i:i+chunk_size])
474 if chunk.strip():
475 all_chunks.append(chunk)
476 all_metadata.append({
477 'file': filepath,
478 'chunk_start': i,
479 'chunk_end': min(i+chunk_size, len(lines))
480 })
481 except Exception as e:
482 print(f"Skipping {filepath}: {e}")
483 continue
484
485 if not all_chunks:
486 return f"No content could be extracted from {owner}/{repo}"
487
488 # Generate embeddings
489 print(f"Generating embeddings for {len(all_chunks)} chunks...")
490 embeddings = np.array([embed_text(chunk) for chunk in all_chunks])
491
492 # Store dimension
493 index_dimension = embeddings.shape[1]
494
495 # Create FAISS index
496 index = faiss.IndexFlatL2(index_dimension)
497 index.add(embeddings.astype('float32'))
498
499 # Store metadata
500 metadata = all_metadata
501
502 return f"✓ Successfully indexed {len(all_chunks)} code chunks from {len(code_files)} files. Index dimension: {index_dimension}"
503
504 except Exception as e:
505 return f"Error ingesting repo: {str(e)}"

508@mcp.tool(name="Semantic_Search")
509async def semanticSearch(query: str, top_k: int = 5) -> str:
510 """
511 Perform semantic search on the data after vectorization and indexing.
512 """
513 global index, metadata, index_dimension
514
515 # Check if index exists
516 if index is None or len(metadata) == 0:
517 return "⚠️ No index found. Please run 'Ingest_Repo_Code' first to build the index."
518
519 try:
520 # Generate query embedding
521 qv = embed_text(query)
522
523 # Verify dimension match
524 if len(qv) != index_dimension:
525 return f"⚠️ Dimension mismatch: Query has {len(qv)} dimensions but index expects {index_dimension}. Please rebuild the index."
526
527 # Reshape for FAISS
528 qv = qv.reshape(1, -1).astype('float32')
529
530 # Search
531 D, I = index.search(qv, min(top_k, len(metadata)))
532
533 # Format results
534 results = []
535 for rank, (dist, idx) in enumerate(zip(D[0], I[0]), 1):
536 if idx < len(metadata):
537 meta = metadata[idx]
538 results.append(
539 f"\n{rank}. File: {meta['file']}\n"
540 f" Lines: {meta['chunk_start']}-{meta['chunk_end']}\n"
541 f" Distance: {dist:.4f}"
542 )
543
544 if not results:
545 return "No results found."
546
547 return f"Search results for: '{query}'\n" + "\n".join(results)
548
549 except AssertionError as e:
550 return f"⚠️ Dimension mismatch error. Index dimension: {index_dimension}, Query dimension: {len(embed_text(query))}. Please rebuild the index."
551 except Exception as e:
552 return f"⚠️ Search error: {str(e)}"

554@mcp.tool(name="Get_File_Functions")
555async def getFileFunctions(owner: str, repo: str, filepath: str, branch: str = "main") -> List[str]:
556 """
557 The MCP tool getFileFunctions extracts Python function definitions from a given file in a GitHub repository. Here's what it does in detail:
558
559 It fetches the raw source code of the specified file from the GitHub repo using fetch_raw_file.
560
561 If the file content is empty or unavailable, it returns an empty list.
562
563 Otherwise, it calls parse_python_functions_ast, which uses Python's built-in ast module to:
564
565 Parse the source code into an abstract syntax tree (AST).
566
567 Traverse the AST to find all function definitions (ast.FunctionDef nodes).
568
569 For each function found, it creates a simple signature string like def function_name(arg1, arg2):.
570
571 The tool returns a list of these function signature strings
572 Purpose:
573 This tool helps junior developers (or any users) quickly understand the structure of Python files by listing the function definitions and their arguments without needing to manually inspect the file line-by-line. It supports the larger goal of the MCP-powered Repo Analyzer to provide guided navigation and understanding of a codebase
574 """
575 code = await fetch_raw_file(owner, repo, filepath, branch)
576 if not code:
577 return []
578 funcs = parse_python_functions_ast(code)
579 return funcs

915async def split_python_functions(file_content: str) -> List[str]:
916 """Parse Python code using AST."""
917 tree = ast.parse(file_content)
918 funcs = []
919 lines = file_content.split('\n')
920
921 for node in ast.iter_child_nodes(tree):
922 if isinstance(node, (ast.FunctionDef, ast.ClassDef)):
923 start_line = node.lineno - 1
924 end_line = node.end_lineno if node.end_lineno else start_line + 1
925 chunk = '\n'.join(lines[start_line:end_line])
926 funcs.append(chunk)
927
928 return funcs

932 """Simple regex-based extraction for JavaScript/TypeScript."""

939 r'class\s+\w+\s*(?:extends\s+\w+\s*)?\{[^}]*\}'

945 funcs.extend([match.group(0) for match in matches])
