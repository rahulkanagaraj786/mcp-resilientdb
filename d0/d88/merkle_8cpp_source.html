27std::string TrverseMakeHash(const BatchClientTransactions& transaction,
28 int l_idx, int r_idx) {
29 if (l_idx == r_idx) {
30 return GetHashValue(transaction.transactions(l_idx).transaction_data());
31 }
32 int mid = (l_idx + r_idx) >> 1;
33
34 std::string l_chd = TrverseMakeHash(transaction, l_idx, mid);
35 std::string r_chd = TrverseMakeHash(transaction, mid + 1, r_idx);
36 return GetHashValue(l_chd + r_chd);
37}

41HashValue Merkle::MakeHash(const BatchClientTransactions& transaction) {
