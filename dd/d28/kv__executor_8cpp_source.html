45std::unique_ptr<std::string> KVExecutor::ExecuteRequest(
46 const google::protobuf::Message& request) {
47 KVResponse kv_response;
48 const KVRequest& kv_request = dynamic_cast<const KVRequest&>(request);
49
50 if (kv_request.cmd() == KVRequest::SET) {
51 Set(kv_request.key(), kv_request.value());
52 } else if (kv_request.cmd() == KVRequest::GET) {
53 kv_response.set_value(Get(kv_request.key()));
54 } else if (kv_request.cmd() == KVRequest::GETALLVALUES) {
55 kv_response.set_value(GetAllValues());
56 } else if (kv_request.cmd() == KVRequest::GETRANGE) {
57 kv_response.set_value(GetRange(kv_request.key(), kv_request.value()));
58 } else if (kv_request.cmd() == KVRequest::SET_WITH_VERSION) {
59 SetWithVersion(kv_request.key(), kv_request.value(), kv_request.version());
60 } else if (kv_request.cmd() == KVRequest::GET_WITH_VERSION) {
61 GetWithVersion(kv_request.key(), kv_request.version(),
62 kv_response.mutable_value_info());
63 } else if (kv_request.cmd() == KVRequest::GET_ALL_ITEMS) {
64 GetAllItems(kv_response.mutable_items());
65 } else if (kv_request.cmd() == KVRequest::GET_KEY_RANGE) {
66 GetKeyRange(kv_request.min_key(), kv_request.max_key(),
67 kv_response.mutable_items());
68 } else if (kv_request.cmd() == KVRequest::GET_HISTORY) {
69 GetHistory(kv_request.key(), kv_request.min_version(),
70 kv_request.max_version(), kv_response.mutable_items());
71 } else if (kv_request.cmd() == KVRequest::GET_TOP) {
72 GetTopHistory(kv_request.key(), kv_request.top_number(),
73 kv_response.mutable_items());
74 } else if (!kv_request.smart_contract_request().empty()) {
75 std::unique_ptr<std::string> resp =
76 contract_manager_->ExecuteData(kv_request.smart_contract_request());
77 if (resp != nullptr) {
78 kv_response.set_smart_contract_response(*resp);
79 }
80 }
81
82 std::unique_ptr<std::string> resp_str = std::make_unique<std::string>();
83 if (!kv_response.SerializeToString(resp_str.get())) {
84 return nullptr;
85 }
86
87 return resp_str;
88}

94
95 if (!kv_request.ParseFromString(request)) {

99

104 } else if (kv_request.cmd() == KVRequest::GETALLVALUES) {

106 } else if (kv_request.cmd() == KVRequest::GETRANGE) {

108 } else if (kv_request.cmd() == KVRequest::SET_WITH_VERSION) {

110 } else if (kv_request.cmd() == KVRequest::GET_WITH_VERSION) {

121 } else if (kv_request.cmd() == KVRequest::GET_TOP) {

169void KVExecutor::GetKeyRange(const std::string& min_key,
170 const std::string& max_key, Items* items) {
171 const std::map<std::string, std::pair<std::string, int>>& ret =
172 storage_->GetKeyRange(min_key, max_key);
173 for (auto it : ret) {
174 Item* item = items->add_item();
175 item->set_key(it.first);
176 item->mutable_value_info()->set_value(it.second.first);
177 item->mutable_value_info()->set_version(it.second.second);
178 }
179}
