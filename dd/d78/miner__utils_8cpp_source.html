29HashValue DigestToHash(const std::string& value) {
30 HashValue hash;
31 const char* pos = value.c_str();
32 for (int j = 0; j < 4; ++j) {
33 uint64_t v = 0;
34 for (int i = 56; i >= 0; i -= 8) {
35 uint64_t tmp = *(pos++) & 0xff;
36 v |= tmp << i;
37 }
38 hash.add_bits(v);
39 }
40 return hash;
41}

43std::string GetHashDigest(const HashValue& hash) {
44 if (hash.bits().empty()) {
45 return "";
46 }
47 char mem[32];
48 int mem_idx = 0;
49 for (int i = 0; i < 4; ++i) {
50 uint64_t bit = hash.bits(i);
51 for (int j = 56; j >= 0; j -= 8) {
52 mem[mem_idx++] = (bit >> j) & 0xff;
53 }
54 }
55 return std::string(mem, 32);
56}

58std::string GetDigestHexString(const std::string digest) {
59 std::string hex;

61 boost::format fmt("%02x");
62 fmt % ((uint32_t)digest[i] & 0xff);
63 hex += fmt.str();

68bool IsValidDigest(const std::string& digest, uint32_t difficulty) {
69 uint32_t num = 0;
70 for (size_t i = 0; i < digest.size(); ++i) {
71 int zeros = 8;
72 if (digest[i] == 0) {
73 } else {
74 uint8_t m = digest[i];
75 while (m > 0) {
76 zeros--;
77 m >>= 1;
78 }
79 }
80 num += zeros;
81 if (zeros != 8) break;
82 }
83 // LOG(ERROR)<<"digest:"<<digest<<" num:"<<num;
84 return num >= difficulty;
85}

92int CmpHash(const HashValue& h1, const HashValue& h2) {
93 if (h1.bits_size() != 4 || h2.bits_size() != 4) return -1;
94 for (int i = 0; i < 4; ++i) {
95 if (h1.bits(i) != h2.bits(i)) {
96 return h1.bits(i) < h2.bits(i) ? -1 : 1;
97 }
98 }
99 return 0;
100}
